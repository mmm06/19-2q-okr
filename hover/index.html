<div class="hover06">hover me</div>

CSS

.hover06 {
    cursor: pointer;
    width: 200px;
    color: #4682b4;
    font-size: 18px;
    font-weight: bold;
    text-align: center;
    padding: 15px;
    position: relative;
    margin: 0 auto;
}
.hover06::after,
.hover06::before {
    content: '';
    display: block;
    position: absolute;
    width: 20%; /* ホバー前の横線の長さになる */
    height: 30%;    /* ホバー前の縦線の長さになる */
    transition: .6s;    /* 変化スピード */
    border-radius: 2px;
}
.hover06::after {
    bottom: 0;
    right: 0;   /* 右下に設置 */
    border-bottom: solid 2px #4682b4;
    border-right: solid 2px #4682b4;    /* 要素の右と下の線のみ表示 */
}
.hover06::before {
    top: 0;
    left: 0;   /* 左上に設置 */
    border-top: solid 2px #4682b4;
    border-left: solid 2px #4682b4;    /* 要素の左と上の線のみ表示 */
}
.hover06:hover::after,
.hover06:hover::before {
    width: 106%;
    height: 120%;   /* 要素を拡大することで線が伸びているように見せる */
}
